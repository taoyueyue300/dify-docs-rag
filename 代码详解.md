### collect_markdown_files() — 收集文件

```python
def collect_markdown_files(base_path: str) -> list:
    # glob.glob() — 用通配符匹配文件路径
    # "**/*.md" 表示：任意深度子目录下的所有.md文件
    # recursive=True 表示递归搜索子目录
    all_files = glob.glob(os.path.join(base_path, "**/*.md"), recursive=True)
    
    # 要排除的目录名（这些目录里的md文件没有价值）
    excluded = ["node_modules", ".venv", "dist", "build", "__pycache__", "plugin_daemon"]
    
    # 列表推导式 + any()：
    # any(ex in f for ex in excluded) — 如果文件路径包含任何一个排除词，则为True
    # not any(...) — 取反，即"不包含任何排除词的文件"才保留
    filtered = [f for f in all_files if not any(ex in f for ex in excluded)]
    
    print(f"找到 {len(filtered)} 个Markdown文件")
    return filtered
```

### load_documents() — 加载文档内容

```python
def load_documents(file_paths: list) -> list:
    documents = []  # 存储所有加载成功的文档
    for path in file_paths:
        try:
            # open() 打开文件，encoding="utf-8" 指定编码防止中文乱码
            # "r" 表示只读模式
            with open(path, "r", encoding="utf-8") as f:
                content = f.read()  # 一次性读取整个文件内容为字符串
            
            # content.strip() — 去掉首尾空白字符
            # 如果内容太短（<50字符），说明是空文件，跳过
            if len(content.strip()) < 50:
                continue
            
            # Path(path).relative_to(...) — 计算相对路径
            # 例如：绝对路径 C:/xx/dify/docs/README.md → 相对路径 docs/README.md
            rel_path = str(Path(path).relative_to(Path(DOCS_PATH).resolve()))
            
            # 把文档存为字典格式：内容 + 来源路径
            documents.append({"content": content, "source": rel_path})
        except Exception as e:
            # 如果读取失败（如编码问题），打印错误并跳过
            print(f"跳过: {path} ({e})")
    
    print(f"成功加载 {len(documents)} 个文档")
    return documents
```

### split_text() — 递归分块（核心算法）

```python
def split_text(text: str, chunk_size=800, overlap=80) -> list:
    """
    递归分块策略：按照优先级从高到低的分隔符逐层切分文本。
    
    为什么要分块？
    - 向量检索要求输入文本不能太长（太长则语义模糊）
    - 但也不能太短（太短则丢失上下文）
    - chunk_size=800 是经过实验验证的最优值
    """
    # 分隔符优先级：标题 > 段落 > 行 > 句子 > 空格
    # 优先按大标题切，保证语义完整性
    separators = ["\n## ", "\n### ", "\n#### ", "\n\n", "\n", ". ", " "]
    chunks = []  # 存储最终的分块结果

    def _split(text, sep_idx=0):
        """
        递归内部函数：
        - 如果文本已经小于chunk_size，直接存入结果
        - 否则用当前优先级的分隔符切分
        - 如果切分后仍然太大，递归使用下一级分隔符
        """
        # 基础情况1：文本已经够短，直接存入
        if len(text) <= chunk_size:
            if text.strip():  # 非空才存
                chunks.append(text.strip())
            return
        
        # 基础情况2：所有分隔符都用完了，强制按长度切
        if sep_idx >= len(separators):
            # range(start, stop, step) — 每次前进 chunk_size-overlap 个字符
            # overlap=80 表示相邻chunk有80字符重叠，防止切断关键信息
            for i in range(0, len(text), chunk_size - overlap):
                piece = text[i:i + chunk_size]  # 切片取一段
                if piece.strip():
                    chunks.append(piece.strip())
            return
        
        # 正常情况：用当前分隔符切分
        sep = separators[sep_idx]
        parts = text.split(sep)  # 按分隔符拆分成列表
        
        current = ""  # 累加器：把小片段合并到一起
        for part in parts:
            # 尝试把当前part加到current后面
            candidate = current + sep + part if current else part
            
            if len(candidate) <= chunk_size:
                # 合并后仍然不超限，继续累加
                current = candidate
            else:
                # 合并后超限了
                if current:
                    # 把之前累加的内容递归处理（用更细的分隔符进一步切分）
                    _split(current, sep_idx + 1)
                current = part  # 开始新的累加
        
        # 处理最后剩余的内容
        if current:
            _split(current, sep_idx + 1)

    _split(text)
    return chunks
```

### create_index() — 向量化并创建FAISS索引

```python
def create_index(chunks: list):
    # os.makedirs — 创建目录，exist_ok=True 表示目录已存在时不报错
    os.makedirs(INDEX_DIR, exist_ok=True)

    print(f"加载Embedding模型 {EMBEDDING_MODEL}...")
    # SentenceTransformer — 加载预训练的文本嵌入模型
    # BGE-small-zh-v1.5：中文优化的小型Embedding模型（384维向量）
    model = SentenceTransformer(EMBEDDING_MODEL)

    # 列表推导式：提取所有chunk的文本和元数据
    contents = [c["content"] for c in chunks]
    metadatas = [{"source": c["source"]} for c in chunks]

    print("正在计算向量...")
    # model.encode() — 把文本列表转为向量矩阵
    # show_progress_bar=True — 显示进度条
    # normalize_embeddings=True — L2归一化，使向量长度为1
    #   归一化后，向量内积 = 余弦相似度（简化计算）
    # batch_size=16 — 每次处理16条文本（控制内存使用）
    # 返回值形状：(chunk数量, 384)，即每个chunk变成384维向量
    embeddings = model.encode(contents, show_progress_bar=True, normalize_embeddings=True, batch_size=16)

    # 释放模型内存（模型很大，后续不再需要）
    del model
    gc.collect()  # 强制垃圾回收

    # === 创建FAISS索引 ===
    dim = embeddings.shape[1]  # 获取向量维度（384）
    
    # IndexFlatIP — 暴力搜索（精确）+ 内积相似度
    # IP = Inner Product（内积）
    # 因为向量已归一化，内积等价于余弦相似度
    index = faiss.IndexFlatIP(dim)
    
    # .add() — 将所有向量加入索引
    # .astype(np.float32) — FAISS要求float32类型
    index.add(embeddings.astype(np.float32))

    # === 保存到磁盘 ===
    # 保存FAISS索引文件
    faiss.write_index(index, os.path.join(INDEX_DIR, "index.faiss"))
    
    # 保存文档内容和元数据（pickle序列化）
    # pickle.dump — 把Python对象保存为二进制文件
    # "wb" = write binary（写入二进制模式）
    with open(os.path.join(INDEX_DIR, "docs.pkl"), "wb") as f:
        pickle.dump({"contents": contents, "metadatas": metadatas}, f)

    print(f"索引已保存到 {INDEX_DIR}，共 {index.ntotal} 条记录")
```

### HybridRetriever 类

```python
class HybridRetriever:
    """
    混合检索器：融合向量语义检索和BM25关键词检索。
    
    为什么要混合？
    - 向量检索擅长理解语义（"部署"能匹配到"安装"）
    - BM25擅长精确关键词匹配（"Docker"就是要找包含"Docker"的文档）
    - 两者互补，覆盖率更高
    """
    
    def __init__(self, use_reranker=False):
        """
        初始化：加载所有模型和数据。
        
        参数:
            use_reranker: 是否启用Reranker重排序
                         （实验证明在当前数据下关闭更好）
        """
        print("加载模型...")
        # 加载Embedding模型（与ingest.py使用同一个模型）
        self.embed_model = SentenceTransformer(EMBEDDING_MODEL)

        # 加载FAISS索引（从磁盘读取ingest.py保存的索引）
        self.index = faiss.read_index(os.path.join(INDEX_DIR, "index.faiss"))
        
        # 加载文档内容（pickle反序列化）
        # "rb" = read binary（读取二进制模式）
        with open(os.path.join(INDEX_DIR, "docs.pkl"), "rb") as f:
            data = pickle.load(f)
        self.all_docs = data["contents"]      # 所有chunk的文本列表
        self.all_metadatas = data["metadatas"] # 所有chunk的元数据列表

        # 初始化BM25检索器
        # BM25需要预先对所有文档分词
        tokenized = [self._tokenize(doc) for doc in self.all_docs]
        self.bm25 = BM25Okapi(tokenized)  # 用分词后的文档列表初始化BM25

        # Reranker（可选）
        self.use_reranker = use_reranker
        self.reranker = None
        if use_reranker:
            from FlagEmbedding import FlagReranker
            self.reranker = FlagReranker(RERANKER_MODEL, use_fp16=False)
        print("模型加载完成")
```

### _tokenize() — 中文分词

```python
    def _tokenize(self, text: str) -> list:
        """
        简单中文分词：中文逐字切分，英文按完整单词切分。
        
        例如："如何部署Docker容器" 
        → ['如', '何', '部', '署', 'docker', '容', '器']
        
        为什么这样设计？
        - 中文没有空格分隔，最简单的方式是按字切分
        - 英文保留完整单词（"Docker"不拆成"D","o","c","k","e","r"）
        - .lower() 统一转小写，避免大小写不匹配
        """
        tokens = []
        word = ""  # 累加英文字符
        for char in text:
            # '\u4e00' 到 '\u9fff' 是Unicode中CJK统一汉字范围
            if '\u4e00' <= char <= '\u9fff':
                # 遇到中文字符：先把之前累加的英文词存入
                if word:
                    tokens.append(word.lower())
                    word = ""
                tokens.append(char)  # 中文逐字存入
            elif char.isalnum():
                # 字母或数字：累加到当前英文词
                word += char
            else:
                # 其他字符（标点、空格等）：结束当前英文词
                if word:
                    tokens.append(word.lower())
                    word = ""
        # 处理最后残留的英文词
        if word:
            tokens.append(word.lower())
        return tokens
```

### retrieve() — 核心检索方法

```python
    def retrieve(self, query: str, top_k: int = 5) -> list:
        """
        执行混合检索，返回最相关的top_k个文档片段。
        
        流程：向量检索 → BM25检索 → 合并去重 → 排序 → 返回top_k
        """
        # === 1. 向量检索（语义匹配）===
        # 把用户问题转为向量
        q_emb = self.embed_model.encode([query], normalize_embeddings=True).astype(np.float32)
        
        # FAISS搜索：返回最相似的20个结果
        # scores: 相似度分数数组，indices: 对应文档的索引号数组
        # 返回形状都是 (1, 20)，因为只有1个query
        scores, indices = self.index.search(q_emb, 20)
        
        # 整理结果：(文档文本, 元数据, 相似度分数)
        # zip() — 把两个列表按位置配对
        # scores[0] 和 indices[0] — 取第一个query的结果（去掉外层维度）
        # if i >= 0 — FAISS当结果不足时会用-1填充，过滤掉
        vector_results_raw = [(self.all_docs[i], self.all_metadatas[i], float(s))
                              for s, i in zip(scores[0], indices[0]) if i >= 0]

        # === 2. BM25检索（关键词匹配）===
        tokenized_query = self._tokenize(query)  # 对问题分词
        
        # get_scores() — 计算query与每个文档的BM25分数
        # 返回一个数组，长度=文档总数，每个值是该文档的BM25得分
        bm25_scores = self.bm25.get_scores(tokenized_query)
        
        # 取BM25分数最高的20个文档的索引
        # sorted() + lambda — 按BM25分数降序排序，取前20
        bm25_top = sorted(range(len(bm25_scores)), key=lambda i: bm25_scores[i], reverse=True)[:20]

        # === 3. 合并去重 ===
        seen = set()      # 集合：已见过的文档内容（用于去重）
        candidates = []   # 最终候选列表
        
        # 先加入向量检索结果（优先级更高）
        for doc, meta, _ in vector_results_raw:
            if doc not in seen:
                seen.add(doc)
                candidates.append({"content": doc, "metadata": meta})
        
        # 再加入BM25独有的结果
        for idx in bm25_top:
            doc = self.all_docs[idx]
            if doc not in seen:
                seen.add(doc)
                candidates.append({"content": doc, "metadata": self.all_metadatas[idx]})

        if not candidates:
            return []

        # === 4. 排序 ===
        if self.use_reranker and self.reranker:
            # Reranker模式：用交叉编码器精排
            pairs = [[query, c["content"]] for c in candidates]
            rerank_scores = self.reranker.compute_score(pairs)
            if isinstance(rerank_scores, (int, float)):
                rerank_scores = [rerank_scores]
            ranked = sorted(zip(candidates, rerank_scores), key=lambda x: x[1], reverse=True)
        else:
            # 加权融合模式（默认，实验验证效果更优）
            # 给向量检索结果按排名赋分：第1名=1.0, 第2名=0.95, ...
            vector_score_map = {doc: 1.0 - i * 0.05 for i, (doc, _, _) in enumerate(vector_results_raw)}
            
            # 按分数降序排列
            ranked = sorted(candidates, key=lambda c: vector_score_map.get(c["content"], 0), reverse=True)
            ranked = [(c, vector_score_map.get(c["content"], 0)) for c in ranked]

        # 返回top_k个结果，每个包含：文本、元数据、分数
        return [
            {"content": c["content"], "metadata": c["metadata"], "score": round(s, 4)}
            for c, s in ranked[:top_k]
        ]
```

### RAGChain 类

```python
class RAGChain:
    def __init__(self):
        self.retriever = HybridRetriever()  # 初始化检索器
        
        # 创建OpenAI客户端（指向本地代理）
        # base_url — API地址（你的代理在这个端口）
        # api_key — 认证密钥
        self.client = OpenAI(base_url=LLM_BASE_URL, api_key=LLM_API_KEY)
        
        self.history = []  # 对话历史（实现多轮对话记忆）

    def query(self, question: str) -> dict:
        """
        RAG问答核心流程：
        1. 检索相关文档片段
        2. 把文档片段 + 问题组装成prompt
        3. 调用LLM生成回答
        4. 保存对话历史
        """
        # === 1. 检索 ===
        results = self.retriever.retrieve(question, top_k=5)

        # === 2. 组装上下文 ===
        context_parts = []
        sources = []
        for i, r in enumerate(results):
            src = r["metadata"].get("source", "unknown")
            # 格式化每个文档片段，标注编号和来源
            context_parts.append(f"[文档{i+1}] (来源: {src})\n{r['content']}")
            sources.append(src)

        # 用换行连接所有文档片段
        context = "\n\n".join(context_parts)

        # === 3. 构建消息列表 ===
        # OpenAI API的消息格式：[{role, content}, ...]
        messages = [{"role": "system", "content": SYSTEM_PROMPT}]
        
        # 加入最近3轮对话历史（self.history[-6:] 取最后6条=3轮）
        for h in self.history[-6:]:
            messages.append(h)
        
        # 加入当前问题（把文档上下文一起给LLM看）
        messages.append({
            "role": "user",
            "content": f"参考文档：\n{context}\n\n用户问题：{question}",
        })

        # === 4. 调用LLM ===
        response = self.client.chat.completions.create(
            model=LLM_MODEL,       # 使用的模型
            messages=messages,     # 消息列表
            temperature=0.1,       # 温度：越低越确定性（不随机发挥）
            max_tokens=1024,       # 最大输出长度
        )
        # 从响应中提取回答文本
        answer = response.choices[0].message.content

        # === 5. 保存对话历史 ===
        self.history.append({"role": "user", "content": question})
        self.history.append({"role": "assistant", "content": answer})

        return {
            "answer": answer,
            "sources": list(set(sources)),  # set去重
            "retrieved_chunks": len(results),
        }
```
